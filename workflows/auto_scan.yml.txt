name: 台股每日自動掃描雷達

on:
  schedule:
    # 每天台灣時間 16:30 執行 (UTC 08:30)
    - cron: '30 8 * * *'
  workflow_dispatch: # 讓你可以手動按按鈕測試

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - name: 複製程式碼
        uses: actions/checkout@v3

      - name: 安裝 Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.9'

      - name: 安裝必要套件
        run: |
          pip install -r requirements.txt

      - name: 執行全市場掃描程式
        env:
          FUGLE_API_KEY: ${{ secrets.FUGLE_API_KEY }}
        run: python scanner.py # 我們把掃描邏輯獨立出來跑

      - name: 上傳結果
        run: |
          git config --local user.email "action@github.com"
          git config --local user.name "GitHub Action"
          git add result.json
          git commit -m "更新今日黑馬名單" || exit 0
          git push
